-- Enable UUID extension
create extension if not exists "uuid-ossp";

-- =========================
-- USER PROFILES TABLE
-- =========================
create table public.user_profiles (
  id uuid primary key references auth.users(id) on delete cascade,
  full_name text,
  age integer,
  gender text,
  created_at timestamptz default now()
);

-- =========================
-- READINGS TABLE
-- =========================
create table public.readings (
  id bigint generated by default as identity primary key,
  user_id uuid not null references auth.users(id) on delete cascade,
  timestamp bigint not null,
  hr integer not null,
  hrv integer not null,
  bp_sys integer not null,
  bp_dia integer not null,
  confidence double precision not null,
  motion integer not null,
  battery integer not null,
  created_at timestamptz default now(),
  
  unique(user_id, timestamp)
);

-- Index for performance
create index idx_readings_user_timestamp 
on public.readings(user_id, timestamp desc);

-- =========================
-- ENABLE RLS
-- =========================
alter table public.readings enable row level security;
alter table public.user_profiles enable row level security;

-- =========================
-- RLS POLICIES
-- =========================

-- Users can insert their own readings
create policy "Users can insert own readings"
on public.readings
for insert
with check (auth.uid() = user_id);

-- Users can select their own readings
create policy "Users can view own readings"
on public.readings
for select
using (auth.uid() = user_id);

-- Users can update their own readings
create policy "Users can update own readings"
on public.readings
for update
using (auth.uid() = user_id);

-- User profile policies
create policy "Users can manage own profile"
on public.user_profiles
for all
using (auth.uid() = id)
with check (auth.uid() = id);













-- Enable RLS on user_profiles table
ALTER TABLE public.user_profiles ENABLE ROW LEVEL SECURITY;

-- Drop existing policies (if any)
DROP POLICY IF EXISTS "Users can manage own profile" ON public.user_profiles;

-- Create comprehensive policy for all operations
CREATE POLICY "Users can manage own profile"
ON public.user_profiles
FOR ALL
TO authenticated
USING (auth.uid() = id)
WITH CHECK (auth.uid() = id);
